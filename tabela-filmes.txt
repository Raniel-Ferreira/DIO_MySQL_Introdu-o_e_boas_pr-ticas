CREATE DATABASE movies-controll;

CREATE TABLE movies(
	id INT PRIMARY KEY AUTO_INCREMENT,
	type INT,
	name VARCHAR 30: NOT NULL,
	total_ep INT,
	atual_ep INT,
	last_view DEFAULT DATE CURRENTE_TIMESTAMP
)
